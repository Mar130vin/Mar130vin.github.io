# TS:发布订阅

interface Evenet {
  on:(name: string, callback:Function) => void
  emit:(name:string,...arg:Array<any>) => void
  off:(name: string, callback:Function) => void
  once:(name: string, callback: Function) => void
}
interface List {
  [key:string]:Array<Function>
}
class Dispatch implements Evenet {
  list:List
  constructor() {
    this.list = {}
  };
    // 监听
  on(name: string, callback:Function) {
    const callbackList: Array<Function> = this.list[name] || [];
    callbackList.push(callback)
    this.list[name] = callbackList
  };

    // 派发
  emit (name:string,...args:Array<any>)  {
    let eventName = this.list[name]
    if (eventName) {
      eventName.forEach(fn => {
        fn.apply(this,args)
      })
    } else {
      console.log('事件未监听'+name);
    }
  };
  off(name: string, callback: Function) {
    let eventName = this.list[name]
    if (eventName && callback) {
      let index = eventName.findIndex(delFn => delFn === callback)
      eventName.splice(index,1)
    }else {
      console.log('事件不存在'+name);
    }
  };
  once (name: string, callback: Function) {
    let decor = (...args: Array<any>) => {
      callback.apply(this, args)
      this.off(name, decor)
    }
    //先走这里
      this.on(name, decor)
  };
}

const o = new Dispatch()
console.log(o);
let fn1 = (...arg:Array<any>) => {
  console.log(arg,1);
}
let fn2 = (...arg:Array<any>) => {
  console.log(arg,2);
}
o.on('post1',fn1)
o.on('post2', fn1)
o.off('post2',fn2)
o.on('post2',fn2)
o.emit('post1','ymf',1,{name:'凡凡'})
o.emit('post2', 'lxy', 2, { name: '雪雪' })

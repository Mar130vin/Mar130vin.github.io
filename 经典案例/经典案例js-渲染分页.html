# 经典案例-渲染分页

首先渲染文章列表

//初始化分页数据
let q = {
  pagenum: 1, // 代表当前页码
  pagesize: 2, // 代表每页显示多少条数据  
}

// 获取文章列表 ----------------------
getArticleList()
getArticleList(){
    //发送请求
    axios({
    method: 'GET',
    url: '/my/xxxxxxx/list',
    params: q,
  }).then(res => {
    console.log(res);
    // 1.当pagenum为最后一页时 且最后一页的数据被删除（最后一页被删的没有数据了，后端返回的最后一页数据为空），此时需要更新列表
        //并且此时pagenum光标还停留在最后一页，就让q.pagenum--，重新获取数据
    // 如果返回的数据为空，而且页码不是1，就让q.pagenum--，重新获取数据
    if (res.data.data.length === 0 && q.pagenum > 1) {
      q.pagenum--
      getArticleList()
    }

    // 把文章总数保存到total这个全局变量里
    total = res.data.total
    // 把当前请求回来的文章列表保存到article_arr这个全局变量里
    article_arr = res.data.data
        
         // 渲染表格 ---------------------------------
        //代码不用管反正该渲染列表
    let trs_arr = res.data.data.map(item => {
      return `
      <tr>
        <td>${item.title}</td> //标题
        <td>${item.cate_name}</td> //分类
        <td>${item.pub_date}</td>//时间
        <td>${item.state}</td>//状态
        <td>
          <button type="button" class="layui-btn layui-btn-xs edit" data-id="${item.Id}"> //
            编辑
          </button>
          <button
            type="button"
            class="layui-btn layui-btn-danger layui-btn-xs delete" data-id="${item.Id}"
          >
            删除
          </button>
        </td>
      </tr>
      `
    })
    $('tbody').html(trs_arr.join(''))
}

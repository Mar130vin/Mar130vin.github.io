<div class="right">

    <!-- 预览图 预览图是div！！！！！cropper会自动在div中生成canvas-->

# js

const image = document.getElementById('image')
const cropper = new Cropper(image, {
  aspectRatio: 20/14,
  crop(event) {
    // 1. 获取裁剪框中的图片 （canvasData的值是一个<canvas>图片标签, 我们完全可以把<canvas>当成<img>）
    const canvasData = cropper.getCroppedCanvas({
      width: 200, //图片宽度
      height: 140,
    })
    $('#preview_image').html(canvasData)
    获取完成后要放到div盒子里面
  },
});
//通过按钮触发input框，如果觉得input框好看的可以不需要这步，尽量取消按钮的默认事件，按钮可能在form标签中会刷新页面
$('.upload_img').on('click',function(e){
  e.preventDefault();
  $('#img_input').click()
})

//input触发条件是change
$('#img_input').on('change',function(e){
  // this.files[0] 用户选择的第一张图片, 不是图片的url地址，但是<img> 里的src属性要的是一个图片的url地址，所以我们需要通过URL.createObjectURL() 这个方法把图片转成成一个url地址
  let img_url = URL.createObjectURL(this.files[0])
  // 错误：由于这个<img>已经被cropper这个插件所控制，所以更改<img>的src值无用
  // $('#avatar_big_img').attr('src', img_url)
  // 正确：替换裁剪区的图片地址
  cropper.replace(img_url)
})

# vue2 cropper

# 安装

npm install cropperjs

# 组件中配置

import 'cropperjs/dist/cropper.css'
import Cropper from 'cropperjs'

# 在组件实例中应用

<template>
  <div class="updata-photo">
    <img :src="userPhoto" ref="img"/>
    <div class="toolbar">
      <span class="cancel" @click="$emit('close')">取消</span>
      <span class="confirm" @click="confirm">完成</span>
    </div>
  </div>
</template>

<script>
import { updateUserPhotoAPI } from '../../../api/index.js'
import 'cropperjs/dist/cropper.css'
import Cropper from 'cropperjs'
export default {
  name: 'UserPhoto',
  model: {
    prop: 'userPhoto',
    event: 'UpdatePhoto'
  },
  props: {
    userPhoto: {
      type: String,
      default: ''
    }
  },
  data () {
    return {
      cropper: null
    }
  },
  methods: {
    initCropper () {
      // 获取图片页面元素
      const image = this.$refs.img
      this.cropper = new Cropper(image, {
        viewMode: 1, // 定义裁剪模式，0：裁剪框可以扩展到画布外， 1,2,3会将裁剪框限制在画布内
        dragMode: 'move', // 拖动模式 move: 移动画布 crop 截图模式
        aspectRatio: 1, // 截图比例, 1:1可以简写为1
        autoCropArea: 1, // 自动截图区
        cropBoxMovable: false, // 截图区域是否可以移动
        cropBoxResizable: false, // 截图区的缩放功能
        background: false, // 是否显示默认背景
        movable: true // 背景图片是否可以移动，默认true
        // crop (event) {
        //   // 当移动图片是会返回的各种数据（后端处理数据时用到）
        //   console.log('x轴', event.detail.x)
        //   console.log('y轴', event.detail.y)
        //   console.log('长度', event.detail.width)
        //   console.log('宽度', event.detail.height)
        //   console.log('旋转角', event.detail.rotate)
        //   console.log('滚动x轴距离', event.detail.scaleX)
        //   console.log('滚动y轴距离', event.detail.scaleY)
        // }
      })
    },
    //confirm () {
  	//	console.log(this.cropper.getData())
    //  处理数据（把数据传递给后端，后端全权处理）
    //},
    confirm () {
      // console.log(this.cropper.getData())
      // this.cropper.getCroppedCanvas()这里生成的是canvas对象
      this.cropper.getCroppedCanvas().toBlob(blob => {
        // 这里是getCroppedCanvas()中的blob
        // 使用浏览器的原生 canvas.toBlob  API来做压缩工作，这意味着它是有损压缩。为了获得更好的图像质量，可以将原始图像和裁剪后的数据上传到服务器，并在服务器上进行裁剪工作。
        this.updateUserPhoto(blob)
      })
    },
    async updateUserPhoto (img) {
      try {
        const formDate = new FormData()
        formDate.append('photo', img)
        const { data: res } = await updateUserPhotoAPI(formDate)
        // console.log(res)
        this.$toast.success('更新成功')
        this.$emit('updatePhoto', res.data.photo)
        this.$emit('close')
      } catch (error) {
        this.$toast.fail('更新失败')
      }
    }
  },
  mounted () {
    this.initCropper()
  }
}
</script>

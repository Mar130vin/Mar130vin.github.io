# vue导出清晰pdf

  1. 引入npm包 Jspdf 和 jspdf-autotable

    npm install jspdf
npm install jspdf-autotable

  2. 需要使用的地方引用

    import JsPDF from 'jspdf'
import autoTable from 'jspdf-autotable

  3. 导出图表

    var myChart = this.$echarts.getInstanceByDom(document.getElementById('customChart'))
var url = myChart.getConnectedDataURL({
        pixelRatio: 2, // 导出的图片分辨率比率,默认是1
        backgroundColor: '#fff', // 图表背景色
        excludeComponents: [
          // 保存图表时忽略的工具组件,默认忽略工具栏
          'toolbox'
        ],
        type: 'png' // 图片类型支持png和jpeg
      })
var doc = new JsPDF('p', 'pt')
doc.addImage(url, 5, 5, 600, 300, 'img')
doc.save('chart.pdf')

  4. 引入中文字体包

    找到一种中文字体，将其转为base64格式，将转换的base64格式的字体js文件放在asset/font文件夹中，并引入文件

    import moo from '@/assets/font/font'  //引入字体，中文

  5. 将数据导出为excel图表

     const tableHeader = this.fields.map(x => x.fieldText) //获取表头
      tableHeader.unshift('序号')  //往表头添加序号列
      const datalist = []
      this.records.forEach((item, index) => {
        const dataItem = []
        this.fields.map(x => x.fieldName).forEach(element => {
          dataItem.push(item[element])
        })
        datalist.push([index + 1, ...dataItem])
      })
      var columns = tableHeader  //表头，数据格式为数组
      var rows = datalist    //dayalist为需要导出的数据，格式为数组中包含数组，即[[],[],[],...]
      const doc = new JsPDF('p', 'pt')
      doc.addFileToVFS('pingfang.ttf', moo)   //设置字体，pdfjs不支持中文字体，需要添加
      doc.addFont('pingfang.ttf', 'b', 'normal') 
      doc.setFont('b')
      doc.autoTable({
        styles: { font: 'b', textColor: [0, 0, 0], halign: 'center' }, // 样式,即字体颜色、表头背景颜色，居中等属性
        theme: 'grid', // 主题
        startY: 40, // 距离上边距离
        columns: columns,
        body: rows
      })
      doc.save('excel.pdf')

  6. 将echarts图片和表格同时导出

        outputPdf () {
      var myChart = this.$echarts.getInstanceByDom(document.getElementById('customChart'))
      var url = myChart.getConnectedDataURL({
        pixelRatio: 2, // 导出的图片分辨率比率,默认是1
        backgroundColor: '#fff', // 图表背景色
        excludeComponents: [
          // 保存图表时忽略的工具组件,默认忽略工具栏
          'toolbox'
        ],
        type: 'png' // 图片类型支持png和jpeg
      })
      var doc = new JsPDF('p', 'pt')
      doc.addImage(url, 5, 5, 600, 300, 'img')
      // doc.addPage() 添加这一行会使表格另起一页
      let imagHeight = myChart.getHeight()      //获取图表的高度
      const tableHeader = this.fields.map(x => x.fieldText)
      tableHeader.unshift('序号')
      const datalist = []
      this.records.forEach((item, index) => {
        const dataItem = []
        this.fields.map(x => x.fieldName).forEach(element => {
          dataItem.push(item[element])
        })
        datalist.push([index + 1, ...dataItem])
      })
      var columns = tableHeader
      var rows = datalist
      doc.addFileToVFS('pingfang.ttf', moo)
      doc.addFont('pingfang.ttf', 'b', 'normal')
      doc.setFont('b')
      doc.autoTable({
        styles: { font: 'b', textColor: [0, 0, 0], halign: 'center' }, // 样式
        theme: 'grid', // 主题
        startY: imagHeight - 100, // 距离上边距离，图表高度-100，具体数值视情况而定
        columns: columns,
        body: rows
      })
      doc.save(this.chartCode + '.pdf')
    },

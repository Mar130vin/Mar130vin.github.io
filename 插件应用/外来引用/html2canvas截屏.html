使用html2canvas截图形成缩略图上传至后端

  1. 安装html2canvas npm install html2canvas
  2. 引用html2canvas import html2canvas from 'html2canvas'
  3. 寻找要截图的元素，并给予设置id属性用于寻找该标签（最好使用id属性）
  4. 截图并上传至服务器
async handleSaveLayout () {
      const el = document.getElementById('pictrue')         // 获取需要截图的最大区域
      html2canvas(el, {
        backgroundColor: this.datavLayout.backgroundColor ? 'transparent' : '#000000', //设置背景颜色，这里这样处理是因为`this.datavLayout.backgroundColor`为空或者null时显示颜色为黑色，一般如果需要保持原有颜色使用'transparent'
        // backgroundColor: 'transparent',
        useCORS: true,
        scale: 1,
        // ignoreElements:(element)=>{
        //	写隐藏元素满足的条件
        //   return true
        // }
      }).then((canvas) => {
        const capture = canvas.toDataURL('image/png')
        const uploadData = new FormData()
        uploadData.append('image', this.base64ToFile(capture, this.datavLayout.datavCode))
        // 将截图上传至服务器
        Api.uploadImg(uploadData, {
          'Content-Type': 'multipart/form-data'
        }).then((res) => {
          this.thumbnail = res.data[0].filePath // 保存上传至服务器图片的相对路径
        }).then(() => {
          const comps = []
          this.datavLayout.comps.forEach(function (item, index) {
            comps.push({
              compCode: item.compCode,
              compName: item.compName,
              compType: item.compType,
              prop: JSON.stringify(item.prop),
              data: JSON.stringify(item.data),
              event: JSON.stringify(item.event)
            })
          })
          const data = {
            id: this.datavLayout.id,
            datavCode: this.datavLayout.datavCode,
            width: this.datavLayout.width,
            height: this.datavLayout.height,
            background: this.datavLayout.background,
            backgroundColor: this.datavLayout.backgroundColor == null ? '' : this.datavLayout.backgroundColor,
            comps: comps,
            thumbnail: this.thumbnail
          }
          // 将保存的图片的路径保存至数据库
          Api.saveLayout(data).then((response) => {
            console.log(response)
            if (response.code == 200) {
              this.$notify({
                type: 'success',
                title: this.$t('common.success'),
                message: response.message,
                position: 'bottom-right'
              })
              // this.loadLayout()
            } else {
              this.$notify({
                type: 'fail',
                title: this.$t('common.fail'),
                message: response.message,
                position: 'bottom-right'
              })
            }
          })
        })
      })
    },

        // base64转为文件格式
    base64ToFile (urlData, fileName) {
      let arr = urlData.split(',')
      let mime = arr[0].match(/:(.*?);/)[1]
      let bytes = atob(arr[1])
      let n = bytes.length
      let ia = new Uint8Array(n)
      while (n--) {
        ia[n] = bytes.charCodeAt(n)
      }
      return new File([ia], fileName, { type: mime })
    },

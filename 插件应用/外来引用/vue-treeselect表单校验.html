<!-- prop属性必须和treeselect的v-model属性值相同 -->
<el-form-item
  :label=" $t('user.dept')"
  prop="departId"         
  label-width="100px"
>
  <el-col :span="24">
    <Treeselect
      v-model="form.departId"
      :options="options"
      :show-count="true"
      :normalizer="normalizer"
      :placeholder="$t('user.deptInfo')"
      @input="changeValue"
    />
  </el-col>
</el-form-item>

// 值发生改变时触发校验规则
changeValue (val) {
  this.$nextTick(() => {
    this.$refs['form'].validateField('departId')
  })
},

// 校验规则
departId: [
  { required: true, type: 'string', message: this.$t('user.deptInfo'), trigger: 'input' }
]

# 注意：设置校验时，必须保证el-form-item 的prop值和treeselect通过v-model绑定的值是一样的，不然校验不通过
